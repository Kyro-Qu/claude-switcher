// ==UserScript==
// @name         Grok SSO Cookie 切换器
// @namespace    http://tampermonkey.net/
// @version      5.9
// @description  统一切号跳转首页，支持自定义默认账号。已移除硬编码 Token，默认使用列表中第一个账号，可手动修改默认值。
// @author       froger
// @match        https://grok.com/*
// @match        https://*.grok.com/*
// @grant        GM_setValue
// @grant        GM_getValue
// @grant        GM_cookie
// @grant        GM_setClipboard
// @grant        GM_xmlhttpRequest
// @grant        GM_registerMenuCommand
// @connect      *
// @run-at       document-start
// ==/UserScript==

(function(factory) {
    if (typeof module !== 'undefined' && module.exports) {
        module.exports = factory;
    }
    if (typeof window !== 'undefined' && typeof document !== 'undefined') {
        factory().run();
    }
})(function createSwitcherApp(runtime = {}) {

    const constants = {
        // 不再包含硬编码的默认 SSO，改为用户自定义存储
        BACKUP_FILENAME: 'grok_sso_backup.json',
        HOME_URL: 'https://grok.com/'
    };

    const gm = {
        getValue: runtime.gm?.getValue || (typeof GM_getValue === 'function' ? GM_getValue : ((_key, fallback) => fallback)),
        setValue: runtime.gm?.setValue || (typeof GM_setValue === 'function' ? GM_setValue : (() => {})),
        setClipboard: runtime.gm?.setClipboard || (typeof GM_setClipboard === 'function' ? GM_setClipboard : (() => {})),
        xmlhttpRequest: runtime.gm?.xmlhttpRequest || (typeof GM_xmlhttpRequest === 'function' ? GM_xmlhttpRequest : null),
        registerMenuCommand: runtime.gm?.registerMenuCommand || (typeof GM_registerMenuCommand === 'function' ? GM_registerMenuCommand : (() => {})),
        cookie: runtime.gm?.cookie || (typeof GM_cookie !== 'undefined' ? GM_cookie : null)
    };

    const documentObj = runtime.document || (typeof document !== 'undefined' ? document : null);
    const windowObj = runtime.window || (typeof window !== 'undefined' ? window : null);
    const locationObj = runtime.location || (typeof location !== 'undefined' ? location : null);
    const confirmFn = runtime.confirm || (typeof confirm === 'function' ? confirm : (() => true));
    const promptFn = runtime.prompt || (typeof prompt === 'function' ? prompt : (() => ''));
    const setTimeoutFn = runtime.setTimeout || (typeof setTimeout === 'function' ? setTimeout : (() => {}));
    const btoaFn = runtime.btoa || (typeof btoa === 'function' ? btoa : null);

    const state = {
        ssoList: [],
        currentIndex: -1,
        webdavConfig: { url: '', user: '', pass: '' }
    };

    // === 默认 SSO 管理 ===
    function getCustomDefaultSSO() {
        const val = gm.getValue('custom_default_sso', '');
        return (typeof val === 'string' && val.length > 0) ? val : null;
    }

    function setCustomDefaultSSO(token) {
        if (token && typeof token === 'string' && token.length > 20) {
            gm.setValue('custom_default_sso', token);
            toast('✅ 自定义默认账号已保存', 'success');
        } else if (token === null || token === '') {
            gm.setValue('custom_default_sso', '');
            toast('🗑️ 已清除自定义默认账号，将使用列表第一个', 'info');
        } else {
            toast('❌ 无效的 SSO Token', 'error');
        }
        updateUI(); // 更新显示
    }

    function getEffectiveDefaultSSO() {
        const custom = getCustomDefaultSSO();
        if (custom) return custom;
        // 无自定义时，使用列表中的第一个
        if (state.ssoList.length > 0) return state.ssoList[0];
        return null; // 列表为空，无默认
    }

    // 加载状态
    function loadState() {
        try {
            const savedList = gm.getValue('sso_list', null);
            state.ssoList = Array.isArray(savedList) ? savedList.filter((token) => typeof token === 'string' && token.length > 20) : [];

            const savedIndex = Number.parseInt(gm.getValue('sso_current_index', -1), 10);
            if (!state.ssoList.length) {
                state.currentIndex = -1;
            } else if (Number.isNaN(savedIndex) || savedIndex < -1) {
                state.currentIndex = -1;
            } else if (savedIndex >= state.ssoList.length) {
                state.currentIndex = -1;
            } else {
                state.currentIndex = savedIndex;
            }
        } catch (_error) {
            state.ssoList = [];
            state.currentIndex = -1;
        }

        state.webdavConfig = gm.getValue('webdav_config', { url: '', user: '', pass: '' }) || { url: '', user: '', pass: '' };
    }

    loadState();

    function formatError(error) {
        if (!error) return '未知错误';
        if (typeof error === 'string') return error;
        if (error.message) return error.message;
        return String(error);
    }

    function normalizeIndex(index, total) {
        if (!total) return -1;
        if (index < 0) return total - 1;
        if (index >= total) return 0;
        return index;
    }

    function resolveRelativeIndex(delta) {
        if (!state.ssoList.length) return -1;
        if (state.currentIndex < 0) {
            return delta >= 0 ? 0 : state.ssoList.length - 1;
        }
        return normalizeIndex(state.currentIndex + delta, state.ssoList.length);
    }

    function setCurrentIndex(nextIndex) {
        state.currentIndex = Number.isInteger(nextIndex) ? nextIndex : -1;
        gm.setValue('sso_current_index', state.currentIndex);
        updateUI();
    }

    function query(selector) {
        return documentObj ? documentObj.querySelector(selector) : null;
    }

    function byId(id) {
        return documentObj ? documentObj.getElementById(id) : null;
    }

    function toast(message, type = 'info') {
        const el = query('.g-toast');
        if (!el) return;
        el.textContent = `${type === 'success' ? '✅' : (type === 'error' ? '❌' : 'ℹ️')} ${message}`;
        el.style.borderLeft = type === 'error' ? '4px solid #ff5252' : '4px solid #4CAF50';
        el.classList.add('show');
        setTimeoutFn(() => el.classList.remove('show'), 3000);
    }

    function showLoading(isLoading) {
        const el = query('.panel-title');
        if (!el) return;
        el.innerHTML = isLoading
            ? '<span class="loading-spin">🔄</span> 处理中...'
            : '<span style="background:#4CAF50; width:8px; height:8px; border-radius:50%; display:inline-block;"></span> Grok 账号助手';
    }

    function updateUI() {
        const idxEl = byId('disp-idx');
        const totalEl = byId('disp-total');

        if (idxEl) {
            if (state.currentIndex >= 0) {
                idxEl.textContent = String(state.currentIndex + 1);
                idxEl.style.color = '#4CAF50';
            } else {
                // 显示默认状态，并标注是否自定义
                const custom = getCustomDefaultSSO();
                if (custom) {
                    idxEl.textContent = '默认(自定义)';
                } else {
                    idxEl.textContent = '默认(首个)';
                }
                idxEl.style.color = '#FFC107';
            }
        }
        if (totalEl) {
            totalEl.textContent = String(state.ssoList.length);
        }
    }

    function redirectToHome() {
        if (locationObj && typeof locationObj.replace === 'function') {
            locationObj.replace(constants.HOME_URL);
        }
    }

    function setGrokCookie(name, value) {
        return new Promise((resolve, reject) => {
            if (!gm.cookie) return reject('需要篡改猴 Beta 版本');
            if (!value) return reject('Cookie 值为空');

            gm.cookie.delete({ name, domain: '.grok.com', path: '/' }, () => {
                gm.cookie.set({
                    name,
                    value,
                    domain: '.grok.com',
                    path: '/',
                    secure: true,
                    httpOnly: true,
                    sameSite: 'lax',
                    expirationDate: Math.floor(Date.now() / 1000) + 31536000
                }, (error) => {
                    if (error) {
                        reject(error);
                        return;
                    }
                    resolve();
                });
            });
        });
    }

    async function applyTokenAndRedirect(token, nextIndex, successMessage) {
        if (!token) {
            toast('❌ 没有可用的 Token', 'error');
            return false;
        }
        showLoading(true);
        try {
            await Promise.all([
                setGrokCookie('sso', token),
                setGrokCookie('sso-rw', token)
            ]);
            setCurrentIndex(nextIndex);
            toast(successMessage, 'success');
            redirectToHome();
            return true;
        } catch (error) {
            showLoading(false);
            toast(`切换失败: ${formatError(error)}`, 'error');
            return false;
        }
    }

    async function switchToSSO(index) {
        if (!state.ssoList.length) {
            toast('列表为空，请先导入账号！', 'error');
            return false;
        }

        const normalizedIndex = normalizeIndex(index, state.ssoList.length);
        const token = state.ssoList[normalizedIndex];
        if (!token) {
            toast('该位置 Token 无效', 'error');
            return false;
        }

        return applyTokenAndRedirect(token, normalizedIndex, `已切换到第 ${normalizedIndex + 1} 个账号`);
    }

    async function switchRelative(delta) {
        const nextIndex = resolveRelativeIndex(delta);
        if (nextIndex < 0) {
            toast('列表为空，请先导入账号！', 'error');
            return false;
        }
        return switchToSSO(nextIndex);
    }

    async function restoreDefault() {
        const defaultToken = getEffectiveDefaultSSO();
        if (!defaultToken) {
            toast('没有可用的默认账号！请先导入账号或设置自定义默认 SSO', 'error');
            return false;
        }
        if (!confirmFn('恢复默认账号并跳转首页?')) return false;
        return applyTokenAndRedirect(defaultToken, -1, '已恢复默认账号');
    }

    // === 自定义默认 SSO 设置 UI ===
    function showSetDefaultModal() {
        const current = getCustomDefaultSSO() || '';
        const newToken = promptFn('请输入要设为默认的 SSO Token（留空则清除自定义，使用列表第一个）:\n\n当前值：' + (current ? current.substring(0, 30) + '…' : '(无自定义)'), current);
        if (newToken === null) return;
        if (newToken.trim() === '') {
            setCustomDefaultSSO(null);
        } else if (newToken.trim().length > 20) {
            setCustomDefaultSSO(newToken.trim());
        } else {
            toast('Token 长度不足，未保存', 'error');
        }
    }

    const WebDAV = {
        getHeaders() {
            if (!btoaFn) throw new Error('当前环境不支持 Base64 编码');
            return {
                Authorization: `Basic ${btoaFn(`${state.webdavConfig.user}:${state.webdavConfig.pass}`)}`,
                'Content-Type': 'application/json; charset=utf-8'
            };
        },
        getUrl() {
            let url = String(state.webdavConfig.url || '').trim();
            if (!url) return '';
            if (!url.endsWith('/')) url += '/';
            return url + constants.BACKUP_FILENAME;
        },
        fetchRemote() {
            return new Promise((resolve, reject) => {
                if (!state.webdavConfig.url) return reject('请先配置 WebDAV URL');
                if (!gm.xmlhttpRequest) return reject('当前环境不支持 GM_xmlhttpRequest');

                gm.xmlhttpRequest({
                    method: 'GET',
                    url: this.getUrl(),
                    headers: this.getHeaders(),
                    onload: (res) => {
                        if (res.status === 200) {
                            try {
                                resolve(JSON.parse(res.responseText));
                            } catch (_error) {
                                reject('文件格式错误');
                            }
                            return;
                        }
                        if (res.status === 404) {
                            resolve([]);
                            return;
                        }
                        reject(`HTTP ${res.status}`);
                    },
                    onerror: () => reject('网络错误')
                });
            });
        },
        putRemote(data) {
            return new Promise((resolve, reject) => {
                if (!gm.xmlhttpRequest) return reject('当前环境不支持 GM_xmlhttpRequest');

                gm.xmlhttpRequest({
                    method: 'PUT',
                    url: this.getUrl(),
                    headers: this.getHeaders(),
                    data: JSON.stringify(data),
                    onload: (res) => {
                        if (res.status >= 200 && res.status < 300) {
                            resolve();
                            return;
                        }
                        reject(`HTTP ${res.status}`);
                    },
                    onerror: () => reject('上传失败')
                });
            });
        },
        async syncMerge() {
            try {
                toast('☁️ 正在连接云端...', 'info');
                const remoteList = await this.fetchRemote();
                const mergedSet = new Set([...state.ssoList, ...remoteList]);
                state.ssoList = Array.from(mergedSet).filter((token) => token && token.length > 20);
                gm.setValue('sso_list', state.ssoList);
                if (!state.ssoList.length) {
                    setCurrentIndex(-1);
                } else if (state.currentIndex >= state.ssoList.length) {
                    setCurrentIndex(-1);
                } else {
                    updateUI();
                }
                await this.putRemote(state.ssoList);
                toast(`同步完成! 总计: ${state.ssoList.length}`, 'success');
            } catch (error) {
                toast(`同步失败: ${formatError(error)}`, 'error');
            }
        },
        async syncRestore() {
            if (!confirmFn('警告：这将使用云端数据覆盖本地数据，确定吗？')) return;
            try {
                toast('☁️ 正在下载...', 'info');
                const remoteList = await this.fetchRemote();
                if (!remoteList.length) {
                    toast('云端无数据', 'error');
                    return;
                }
                state.ssoList = remoteList.filter((token) => token && token.length > 20);
                gm.setValue('sso_list', state.ssoList);
                setCurrentIndex(-1);
                toast(`恢复成功! 加载 ${state.ssoList.length} 个账号`, 'success');
            } catch (error) {
                toast(`恢复失败: ${formatError(error)}`, 'error');
            }
        }
    };

    function createUI() {
        if (!documentObj || !documentObj.body || !documentObj.head) return;
        if (byId('grok-sso-panel')) return;

        const style = documentObj.createElement('style');

                </div>
                <div class="btn-grid">
                    <button class="g-btn" id="btn-prev">⬅ 上一个</button>
                    <button class="g-btn" id="btn-next">下一个 ➡</button>

                </div>
                <button class="g-btn danger" id="btn-default" style="width:100%">🏠 恢复默认账号</button>
                <div class="toolbar">
                    <button class="tool-icon" id="tool-star" title="设置/清除默认 SSO">⭐</button>
                    <button class="tool-icon" id="tool-sync" title="WebDAV 云同步">☁️</button>

                    <button class="tool-icon" id="tool-export" title="导出备份">📤</button>

        minIcon.id = 'grok-min-icon';

        const modalWebDAV = documentObj.createElement('div');
        modalWebDAV.className = 'g-modal';

            <div class="g-modal-box">

            el.style.top = savedPos.top;
            el.style.left = savedPos.left || 'auto';
            el.style.right = savedPos.right || 'auto';

        applyPosition(minIcon);

        function toggleState(minimized) {
            gm.setValue('is_minimized', minimized);
            if (minimized) {
                minIcon.style.top = panel.style.top;
                minIcon.style.left = panel.style.left;
                minIcon.style.right = panel.style.right;
                panel.style.display = 'none';
                minIcon.style.display = 'flex';
            } else {
                panel.style.top = minIcon.style.top;
                panel.style.left = minIcon.style.left;
                panel.style.right = minIcon.style.right;
                minIcon.style.display = 'none';
                panel.style.display = 'block';
            }
        }

        if (gm.getValue('is_minimized', false)) {
            panel.style.display = 'none';
            minIcon.style.display = 'flex';
        } else {
            minIcon.style.display = 'none';
        }

        gm.registerMenuCommand('🔄 重置面板位置', () => {
            savedPos = { top: '80px', right: '50px' };
            gm.setValue('panel_pos', savedPos);
            gm.setValue('is_minimized', false);
            applyPosition(panel);
            toggleState(false);
            toast('位置已重置', 'success');
        });
        gm.registerMenuCommand('⭐ 设置默认 SSO', () => showSetDefaultModal());

        function enableDrag(element, handle) {
            let isDragging = false;
            let startX = 0;
            let startY = 0;
            let startLeft = 0;
            let startTop = 0;

            handle.addEventListener('mousedown', (event) => {
                isDragging = false;
                element.removeAttribute('data-dragging');

                const rect = element.getBoundingClientRect();
                element.style.right = 'auto';
                element.style.bottom = 'auto';
                element.style.left = `${rect.left}px`;
                element.style.top = `${rect.top}px`;
                startX = event.clientX;
                startY = event.clientY;
                startLeft = rect.left;
                startTop = rect.top;

                documentObj.addEventListener('mousemove', onMouseMove);
                documentObj.addEventListener('mouseup', onMouseUp);
            });

            function onMouseMove(event) {
                if (Math.abs(event.clientX - startX) > 3 || Math.abs(event.clientY - startY) > 3) {
                    isDragging = true;
                    element.setAttribute('data-dragging', 'true');
                    documentObj.body.style.userSelect = 'none';
                }
                if (isDragging) {
                    element.style.left = `${startLeft + event.clientX - startX}px`;
                    element.style.top = `${startTop + event.clientY - startY}px`;
                }
            }

            function onMouseUp() {
                documentObj.removeEventListener('mousemove', onMouseMove);
                documentObj.removeEventListener('mouseup', onMouseUp);
                documentObj.body.style.userSelect = '';
                if (isDragging) {
                    const newPos = { top: element.style.top, left: element.style.left };
                    gm.setValue('panel_pos', newPos);
                    if (element === panel) {
                        minIcon.style.top = newPos.top;
                        minIcon.style.left = newPos.left;
                    } else {
                        panel.style.top = newPos.top;
                        panel.style.left = newPos.left;
                    }
                    setTimeoutFn(() => element.removeAttribute('data-dragging'), 50);
                }
            }
        }

        enableDrag(panel, panel.querySelector('.panel-header'));
        enableDrag(minIcon, minIcon);

        byId('btn-minimize').onclick = () => toggleState(true);
        minIcon.onclick = () => {
            if (!minIcon.hasAttribute('data-dragging')) toggleState(false);
        };

        byId('btn-next').onclick = () => {
            switchRelative(1);
        };
        byId('btn-prev').onclick = () => {
            switchRelative(-1);
        };
        byId('btn-random').onclick = () => {
            if (!state.ssoList.length) {
                toast('列表为空', 'error');
                return;
            }
            switchToSSO(Math.floor(Math.random() * state.ssoList.length));
        };
        byId('btn-default').onclick = () => {
            restoreDefault();
        };
        byId('btn-goto').onclick = () => {
            if (!state.ssoList.length) {
                toast('列表为空', 'error');
                return;
            }
            const nextIndex = Number.parseInt(promptFn(`序号 (1-${state.ssoList.length}):`), 10) - 1;
            if (!Number.isNaN(nextIndex)) {
                switchToSSO(nextIndex);
            }
        };

        // 新增：设置默认 SSO 按钮
        byId('tool-star').onclick = () => showSetDefaultModal();

        byId('tool-sync').onclick = () => modalWebDAV.classList.add('show');
        byId('wd-cancel').onclick = () => modalWebDAV.classList.remove('show');
        byId('wd-save').onclick = () => {
            state.webdavConfig = {
                url: byId('wd-url').value,
                user: byId('wd-user').value,
                pass: byId('wd-pass').value
            };
            gm.setValue('webdav_config', state.webdavConfig);
            toast('已保存', 'success');
        };
        byId('wd-up').onclick = () => WebDAV.syncMerge();
        byId('wd-down').onclick = () => WebDAV.syncRestore();

        const txtArea = byId('import-area');
        byId('tool-import').onclick = () => {
            txtArea.value = '';
            modalImport.classList.add('show');
            txtArea.focus();
        };
        byId('import-cancel').onclick = () => modalImport.classList.remove('show');
        byId('import-run').onclick = () => {
            const raw = txtArea.value;
            if (!raw.trim()) return;

            const lines = raw.split('\n').map((line) => line.trim()).filter((line) => line.length > 20);
            if (!lines.length) {
                toast('未检测到有效数据', 'error');
                return;
            }

            const currentSet = new Set(state.ssoList);
            let addedCount = 0;
            let duplicateCount = 0;

            lines.forEach((token) => {
                if (currentSet.has(token)) {
                    duplicateCount += 1;
                    return;
                }
                currentSet.add(token);
                state.ssoList.push(token);
                addedCount += 1;
            });

            if (addedCount > 0) {
                gm.setValue('sso_list', state.ssoList);
                if (state.currentIndex < 0 && state.ssoList.length === addedCount) {
                    updateUI();
                } else {
                    updateUI();
                }
            }

            modalImport.classList.remove('show');
            toast(`识别: ${lines.length} | 新增: ${addedCount} | 重复: ${duplicateCount} | 总计: ${state.ssoList.length}`, 'success');
        };

        byId('tool-export').onclick = () => {
            gm.setClipboard(state.ssoList.join('\n'));
            toast('已复制备份', 'success');
        };
        byId('tool-copy').onclick = () => {
            let token = null;
            if (state.currentIndex >= 0 && state.ssoList[state.currentIndex]) {
                token = state.ssoList[state.currentIndex];
            } else {
                token = getEffectiveDefaultSSO();
            }
            if (token) {
                gm.setClipboard(token);
                toast('已复制 Token', 'success');
            } else {
                toast('没有可复制的 Token', 'error');
            }
        };

        updateUI();
    }

    function init() {
        if (!documentObj) return;
        if (!documentObj.body || !documentObj.head) {
            setTimeoutFn(init, 50);
            return;
        }
        createUI();
    }

    function run() {
        if (runtime.autoInit === false) return;
        init();
    }

    return {
        constants,
        state,
        switchToSSO,
        switchRelative,
        restoreDefault,
        run,
        _internal: {
            formatError,
            normalizeIndex,
            resolveRelativeIndex,
            getEffectiveDefaultSSO,
            setCustomDefaultSSO
        }
    };
});
